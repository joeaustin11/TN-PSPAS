-- MySQL DDL (copy to create your database) CREATE TABLE Companies ( company_id VARCHAR(10) PRIMARY KEY, company_name VARCHAR(100) NOT NULL, district VARCHAR(50), is_verified BOOLEAN DEFAULT FALSE, status ENUM('active','suspended') DEFAULT 'active' ); CREATE TABLE Company_Score ( score_id INT PRIMARY KEY AUTO_INCREMENT, company_id VARCHAR(10) REFERENCES Companies(company_id), ontime_rate FLOAT, quality_rating FLOAT, feedback_score FLOAT, project_count INT, total_score FLOAT GENERATED ALWAYS AS (0.40*ontime_rate + 0.30*quality_rating + 0.20*feedback_score + 0.10*(project_count*2)) ); CREATE TABLE Problem_Statements ( ps_id VARCHAR(10) PRIMARY KEY, govt_id VARCHAR(10) REFERENCES Government_Users(govt_id), title VARCHAR(200) NOT NULL, budget DECIMAL(12,2), deadline DATE, priority ENUM('high','medium','low'), status ENUM('pending','auction_open','auction_closed') ); CREATE TABLE Bids ( bid_id VARCHAR(10) PRIMARY KEY, company_id VARCHAR(10) REFERENCES Companies(company_id), ps_id VARCHAR(10) REFERENCES Problem_Statements(ps_id), bid_amount DECIMAL(12,2), final_score FLOAT, rank_position INT );
